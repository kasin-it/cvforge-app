## API Integration

All service calls should be wrapped with loading states and error handling:

```typescript
// services/api.ts

export async function analyzeJob(input: { url: string } | { text: string }) {
  const analyzer = new JobPostingAnalyzerService();

  if ('url' in input) {
    return analyzer.analyzeFromUrl(input.url);
  }
  return analyzer.analyzeFromText(input.text);
}

export async function optimizeCV(
  cv: CV,
  job: JobPosting,
  options: { mode: OptimizationMode; context?: string }
) {
  const optimizer = new CvOptimizerService();
  return optimizer.enrich(cv, job, options);
}

export async function renderCV(
  cv: CV,
  options: { template: "modern" | "minimal"; format: "pdf" | "html" }
) {
  const renderer = new CVRendererService();

  if (options.format === "html") {
    return renderer.renderHTML(cv, options.template);
  }

  const pdf = await renderer.renderPDF(cv, { template: options.template });
  await renderer.close();
  return pdf;
}
```
